static main() {
    auto ea;
    auto fp;
    fp = fopen("functions.txt", "w");
    if (fp) {
        for (ea = NextFunction(0); ea != BADADDR; ea = NextFunction(ea)) {
            fprintf(fp, "Function: %s, Start Address: %s, End Address: %s\n", GetFunctionName(ea), atoa(ea), atoa(GetFunctionAttr(ea, FUNCATTR_END)));
        }
        fclose(fp);
        Message("Finished writing to functions.txt\n");
    } else {
        Message("Failed to open file for writing\n");
    }
}
